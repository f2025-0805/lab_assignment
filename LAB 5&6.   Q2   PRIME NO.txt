#include <iostream>
using namespace std;

int main() {
    bool rerun = true;
    while(rerun){
    
    int n;
    bool isPrime = false;
    cout << "Enter a Number: ";
    cin>>n;
    if(n == 2){
        cout<<n<<" is a prime Number!";
    }
    else{
    for(int i=2; i < n; i++){
        if(n%i == 0){
            isPrime = false;
            break;
        }
        else{
            isPrime = true;
        }
    }
    if (isPrime == true){
        cout<<n << " is a prime Number!";
    }
    else{
        cout<<n<<" is a composite number!";
    }
    }
    cout<<"\nEnter 1 to rerun and 0 to exit: ";
    cin>>rerun;
}
    return 0;
}
