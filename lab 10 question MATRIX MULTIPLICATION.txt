#include <iostream>
using namespace std;
int main(){
    int a[3][3],b[3][3],c[3][3];
    int m,n,p,q;
    cout<<"enter the number of rows and columns  for matrix a "<<endl;
    cin>>m>>n;
    cout<<"enter the number of rows and columns for matrix b"<<endl;
    cin>>p>>q;
    
    if(n!=p){
        cout<<" multiplication not possible"<<endl;
        return 0;
    }
    for (int i=0;i<m;i++){
        for(int j=0;j<n;j++){
            cout<<"enter the value for a index:"<<i<<","<<j<<endl;
            cin>>a[i][j];
            cout<<endl;
        }
    }
    for(int i=0;i<p;i++){
        for(int j=0;j<q;j++){
             cout<<"enter the value for b index:"<<i<<","<<j<<endl;
            cin>>b[i][j];
            cout<<endl;
        }
    }
    for(int i=0;i<m;i++){
        for(int j=0;j<q;j++){
            c[i][j]=0;
            for(int k=0;k<n;k++){
                c[i][j]+=a[i][k]*b[k][j];
            }
        }
    }
    cout<<"result"<<endl;
    for(int i=0;i<m;i++){
        for(int j=0;j<q;j++){
         cout<<c[i][j]<<" "; 
          cout<<endl;
        }
    }
    return 0;
}
